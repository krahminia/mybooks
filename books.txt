                    <div class="space-y-3">
                        <template x-for="item in selectedBookData.quotes" :key="item.id">

                        <h4 class="text-sm font-bold text-gray-700 dark:text-gray-200 mb-4 flex items-center"><i class="fa-solid fa-link mr-2 text-blue-400"></i> Referensi & Tautan</h4>
                        <div class="space-y-2">

    <script>
        // Register Plugin ChartJS Datalabels Global
        Chart.register(ChartDataLabels);

        function dashboardApp() {
            return {
                isLoading: true,
                isDarkMode: false, // State for Dark Mode
                currentView: 'dashboard', 
                searchQuery: '',
                
                menus: [
                    { id: 'dashboard', label: 'Dashboard', icon: 'fa-solid fa-chart-pie', title: 'Analitik Pribadi' },
                    { id: 'katalog', label: 'Katalog Buku', icon: 'fa-solid fa-book', title: 'Katalog Kepustakaan' },
                    { id: 'knowledge', label: 'Knowledge Hub', icon: 'fa-solid fa-lightbulb', title: 'Catatan Knowledge Hub & Second Brain' },
                    { id: 'favorit', label: 'Buku Favorit', icon: 'fa-solid fa-star', title: 'Koleksi Buku Favorit' }
                ],

                urls: {
                    books: 'https://docs.google.com/spreadsheets/d/e/2PACX-1vQDHtpn0OjFI5OCKbQyEiIvxzPv8Y925h7neiMib2ByIITDZV22g0YL_s4-oNpy55ZrKl3zTfMi6f8C/pub?gid=0&single=true&output=csv',
                    insights: 'https://docs.google.com/spreadsheets/d/e/2PACX-1vQDHtpn0OjFI5OCKbQyEiIvxzPv8Y925h7neiMib2ByIITDZV22g0YL_s4-oNpy55ZrKl3zTfMi6f8C/pub?gid=712348867&single=true&output=csv',
                    quotes: 'https://docs.google.com/spreadsheets/d/e/2PACX-1vQDHtpn0OjFI5OCKbQyEiIvxzPv8Y925h7neiMib2ByIITDZV22g0YL_s4-oNpy55ZrKl3zTfMi6f8C/pub?gid=1999049270&single=true&output=csv',
                    actions: 'https://docs.google.com/spreadsheets/d/e/2PACX-1vQDHtpn0OjFI5OCKbQyEiIvxzPv8Y925h7neiMib2ByIITDZV22g0YL_s4-oNpy55ZrKl3zTfMi6f8C/pub?gid=399020959&single=true&output=csv',
                    links: 'https://docs.google.com/spreadsheets/d/e/2PACX-1vQDHtpn0OjFI5OCKbQyEiIvxzPv8Y925h7neiMib2ByIITDZV22g0YL_s4-oNpy55ZrKl3zTfMi6f8C/pub?gid=554005956&single=true&output=csv'
                },

                db: { books: [], insights: [], quotes: [], actions: [], links: [] },
                stats: [],
                recentBooks: [],
                unreadBooks: [],

                // Filter States
                filterStatus: 'Semua',
                filterCategory: 'Semua',
                hubCategory: 'Semua',
                hubTag: '',

                // Modal State
                isModalOpen: false,
                selectedBook: null,
                selectedBookData: { insights: [], quotes: [], actions: [], links: [] },

                // Calendar State
                currentDate: new Date(),
                currentMonth: new Date().getMonth(),
                currentYear: new Date().getFullYear(),
                monthNames: ["Januari", "Februari", "Maret", "April", "Mei", "Juni", "Juli", "Agustus", "September", "Oktober", "November", "Desember"],
                calendarCells: [],

                chartInstances: {},

                async init() {
                    // Check local storage for dark mode preference
                    if (localStorage.getItem('theme') === 'dark' || (!('theme' in localStorage) && window.matchMedia('(prefers-color-scheme: dark)').matches)) {
                        this.isDarkMode = true;
                    }

                    try {
                        const fetchCSV = (url) => new Promise((resolve, reject) => {
                            Papa.parse(url, { download: true, header: true, skipEmptyLines: true, complete: resolve, error: reject });
                        });

                        const [b, i, q, a, l] = await Promise.all([
                            fetchCSV(this.urls.books), fetchCSV(this.urls.insights), 
                            fetchCSV(this.urls.quotes), fetchCSV(this.urls.actions), fetchCSV(this.urls.links)
                        ]);
                        // Clean data IDs to trim spaces
                        this.db.books = b.data.map(item => ({...item, id: item.id?.trim()}));
                        this.db.insights = i.data.map(item => ({...item, bookId: item.bookId?.trim()}));
                        this.db.quotes = q.data.map(item => ({...item, bookId: item.bookId?.trim()}));
                        this.db.actions = a.data.map(item => ({...item, bookId: item.bookId?.trim()}));
                        this.db.links = l.data.map(item => ({...item, bookId: item.bookId?.trim()}));

                        this.processDashboardData();
                        this.generateCalendar();
                        
                        this.isLoading = false;
                        
                        this.$watch('currentView', (val) => {
                            if(val === 'dashboard') {
                                this.$nextTick(() => this.initCharts());
                            }
                        });
                        this.$watch('isDarkMode', () => {
                            this.$nextTick(() => { if (this.currentView === 'dashboard') this.initCharts(); });
                        });
                        
                        this.$nextTick(() => { if (this.currentView === 'dashboard') this.initCharts(); });

                    } catch (error) {
                        console.error(error);
                        alert("Gagal memuat data Spreadsheet. Pastikan link ter-publish CSV.");
                    }
                },

                toggleTheme() {
                    this.isDarkMode = !this.isDarkMode;
                    localStorage.setItem('theme', this.isDarkMode ? 'dark' : 'light');
                },

                changeView(view) {
                    this.currentView = view;
                    this.searchQuery = '';
                    // Reset filter form when changing views
                    if(view === 'katalog' || view === 'favorit') {
                        this.filterCategory = 'Semua';
                        this.filterStatus = 'Semua';
                    }
                    window.scrollTo({ top: 0, behavior: 'smooth' });
                },

                isFavorite(b) {
                    const val = (b.favorite || '').toString().trim().toLowerCase();
                    return val === 'yes' || val === 'true' || val === '1' || val === 'y';
                },

                processDashboardData() {
                    const totalBooks = this.db.books.length;
                    const authors = new Set(this.db.books.map(b => b.author)).size;
                    const categories = new Set(this.db.books.map(b => b.category?.trim())).size;
                    const upKontenCount = this.db.links.length;
                    const insightCount = this.db.insights.length; 
                    const favoriteCount = this.db.books.filter(b => this.isFavorite(b)).length;
                    
                    this.stats = [
                        { label: 'TOTAL BUKU', value: totalBooks, icon: 'fa-solid fa-book-open' },
                        { label: 'PENULIS', value: authors, icon: 'fa-solid fa-feather' },
                        { label: 'KATEGORI', value: categories, icon: 'fa-solid fa-tags' },
                        { label: 'UP KONTEN', value: upKontenCount, icon: 'fa-solid fa-arrow-up-right-dots' },
                        { label: 'FAVORITE', value: favoriteCount, icon: 'fa-solid fa-star' },
                        { label: 'INSIGHT', value: insightCount, icon: 'fa-solid fa-lightbulb' }
                    ];
                    
                    this.recentBooks = [...this.db.books].reverse().slice(0, 3);
                    this.unreadBooks = this.db.books.filter(b => {
                        let stat = (b.status || '').toLowerCase().trim();
                        return stat === 'wishlist' || stat === 'reading' || stat === 'membaca';
                    }).reverse();
                },

                // --- DATA COMPUTATION UNTUK VIEWS ---
                get uniqueCategories() {
                    // Perbaikan: Trim spasi supaya tidak terjadi error/bug duplikasi kategori dengan spasi terselip
                    return [...new Set(this.db.books.map(b => b.category?.trim()).filter(Boolean))].sort();
                },

                get uniqueTags() {
                    let tags = new Set();
                    this.db.insights.forEach(i => {
                        if(i.tag) {
                            let parts = i.tag.split(/[\s,]+/);
                            parts.forEach(p => {
                                if(p.trim()) tags.add(p.trim().toUpperCase());
                            });
                        }
                    });
                    return [...tags].sort();
                },

                get filteredKatalog() {
                    let result = this.db.books;
                    if (this.currentView === 'favorit') {
                        result = result.filter(b => this.isFavorite(b));
                    }

                    if (this.filterStatus !== 'Semua') {
                        result = result.filter(b => (b.status || '').toLowerCase() === this.filterStatus.toLowerCase());
                    }
                    if (this.filterCategory !== 'Semua') {
                        // Perbaikan: Trim saat filter
                        result = result.filter(b => b.category && b.category.trim() === this.filterCategory.trim());
                    }
                    if (this.searchQuery) {
                        const q = this.searchQuery.toLowerCase();
                        result = result.filter(b => 
                            (b.title && b.title.toLowerCase().includes(q)) || 
                            (b.author && b.author.toLowerCase().includes(q))
                        );
                    }
                    return result;
                },

                get filteredHub() {
                    let mapped = {};
                    this.db.insights.forEach(insight => {
                        if(this.hubTag && !(insight.tag || '').toUpperCase().includes(this.hubTag)) return;
                        if(this.searchQuery && !(insight.insight || '').toLowerCase().includes(this.searchQuery.toLowerCase())) return;

                        const book = this.db.books.find(b => b.id === insight.bookId);
                        if(!book) return;
                        
                        if(this.hubCategory !== 'Semua' && book.category?.trim() !== this.hubCategory.trim()) return;

                        let cat = book.category?.trim() || 'Uncategorized';
                        if(!mapped[cat]) mapped[cat] = {};
                        if(!mapped[cat][book.id]) mapped[cat][book.id] = { book: book, insights: [] };
                        
                        mapped[cat][book.id].insights.push(insight);
                    });
                    
                    let finalArr = [];
                    for(const [cat, books] of Object.entries(mapped)) {
                        let bookArr = Object.values(books);
                        if(bookArr.length > 0) {
                            finalArr.push({ category: cat, books: bookArr });
                        }
                    }
                    return finalArr.sort((a,b) => a.category.localeCompare(b.category));
                },

                // --- MODAL LOGIC ---
                openModal(book) {
                    this.selectedBook = book;
                    this.selectedBookData.insights = this.db.insights.filter(i => i.bookId === book.id);
                    this.selectedBookData.quotes = this.db.quotes.filter(q => q.bookId === book.id);
                    this.selectedBookData.actions = this.db.actions.filter(a => a.bookId === book.id);
                    this.selectedBookData.links = this.db.links.filter(l => l.bookId === book.id);
                    this.isModalOpen = true;
                    document.body.style.overflow = 'hidden';
                },
                closeModal() {
                    this.isModalOpen = false;
                    setTimeout(() => { this.selectedBook = null; }, 300);
                    document.body.style.overflow = '';
                },

                // --- CALENDAR LOGIC ---
                changeMonth(offset) {
                    this.currentMonth += offset;
                    if (this.currentMonth > 11) { this.currentMonth = 0; this.currentYear++; }
                    if (this.currentMonth < 0) { this.currentMonth = 11; this.currentYear--; }
                    this.generateCalendar();
                },

                generateCalendar() {
                    this.calendarCells = [];
                    const firstDay = new Date(this.currentYear, this.currentMonth, 1).getDay();
                    const daysInMonth = new Date(this.currentYear, this.currentMonth + 1, 0).getDate();
                    let startingDayIndex = firstDay === 0 ? 6 : firstDay - 1;
                    
                    for (let i = 0; i < startingDayIndex; i++) this.calendarCells.push({ date: null, books: [] });
                    for (let d = 1; d <= daysInMonth; d++) {
                        const booksOnThisDay = this.db.books.filter(b => {
                            if (!b.status || (b.status.toLowerCase() !== 'finished' && b.status.toLowerCase() !== 'selesai')) return false;
                            if (!b.dateFinished) return false;
                            const bDate = new Date(b.dateFinished);
                            return bDate.getFullYear() === this.currentYear && bDate.getMonth() === this.currentMonth && bDate.getDate() === d;
                        });
                        this.calendarCells.push({ date: d, books: booksOnThisDay });
                    }
                },

                // --- CHARTS LOGIC ---
                initCharts() {
                    Chart.defaults.font.family = "'Nunito', sans-serif";
                    Chart.defaults.color = this.isDarkMode ? '#9ca3af' : '#6b7280'; // Sesuaikan warna teks chart dengan tema

                    if(this.chartInstances.cat) this.chartInstances.cat.destroy();
                    if(this.chartInstances.stat) this.chartInstances.stat.destroy();

                    // 1. Horizontal Bar Chart (Kategori)
                    const categoryCounts = {};
                    this.db.books.forEach(b => {
                        const cat = b.category?.trim() || 'Lainnya';
                        categoryCounts[cat] = (categoryCounts[cat] || 0) + 1;
                    });
                    const sortedCategories = Object.entries(categoryCounts).sort((a, b) => b[1] - a[1]);

                    const ctxCat = document.getElementById('categoryChart');
                    if(ctxCat) {
                        this.chartInstances.cat = new Chart(ctxCat.getContext('2d'), {
                            type: 'bar',
                            data: {
                                labels: sortedCategories.map(c => c[0]),
                                datasets: [{
                                    data: sortedCategories.map(c => c[1]),
                                    backgroundColor: '#8c7b6c',
                                    borderRadius: 4,
                                    barThickness: 16
                                }]
                            },
                            options: {
                                indexAxis: 'y', 
                                responsive: true, maintainAspectRatio: false,
                                layout: { padding: { right: 30 } },
                                plugins: { 
                                    legend: { display: false },
                                    datalabels: { 
                                        color: this.isDarkMode ? '#e5e7eb' : '#5c4e43',
                                        anchor: 'end', align: 'right', offset: 4,
                                        font: { weight: 'bold', size: 12 }
                                    }
                                },
                                scales: {
                                    x: { display: false, grid: { display: false } }, 
                                    y: { grid: { display: false }, border: {display: false}, ticks: { font: { weight: 'bold' } } }
                                }
                            }
                        });
                    }

                    // 2. Status Chart (Donut)
                    const statusCounts = { Finished: 0, Reading: 0, Wishlist: 0 };
                    this.db.books.forEach(b => {
                        let stat = (b.status || '').toLowerCase().trim();
                        if (stat === 'finished' || stat === 'selesai') statusCounts.Finished++;
                        else if (stat === 'reading' || stat === 'membaca') statusCounts.Reading++;
                        else if (stat === 'wishlist') statusCounts.Wishlist++;
                    });
                    
                    const ctxStat = document.getElementById('statusChart');
                    if(ctxStat) {
                        this.chartInstances.stat = new Chart(ctxStat.getContext('2d'), {
                            type: 'doughnut',
                            data: {
                                labels: ['Finished', 'Reading', 'Wishlist'],
                                datasets: [{
                                    data: [statusCounts.Finished, statusCounts.Reading, statusCounts.Wishlist],
                                    backgroundColor: ['#5c4e43', '#8c7b6c', '#e3dcd2'],
                                    borderColor: this.isDarkMode ? '#1f2937' : '#ffffff',
                                    borderWidth: 2,
                                    cutout: '70%'
                                }]
                            },
                            options: {
                                responsive: true, maintainAspectRatio: false,
                                plugins: {
                                    // Perbaikan: Munculkan Label Persentase di Pie Chart
                                    datalabels: { 
                                        display: true,
                                        color: '#ffffff',
                                        font: { weight: 'bold', size: 11 },
                                        formatter: (value, ctx) => {
                                            let datasets = ctx.chart.data.datasets;
                                            if (datasets.indexOf(ctx.dataset) === datasets.length - 1) {
                                                let sum = datasets[0].data.reduce((a, b) => a + b, 0);
                                                let percentage = Math.round((value / sum) * 100) + '%';
                                                return percentage === '0%' ? '' : percentage; // Sembunyikan label kalau 0%
                                            } else {
                                                return percentage;
                                            }
                                        }
                                    }, 
                                    legend: { position: 'bottom', labels: { usePointStyle: true, padding: 15, font: { weight: 'bold'} } }
                                }
                            }
                        });
                    }
                }
            }
        }
    </script>
